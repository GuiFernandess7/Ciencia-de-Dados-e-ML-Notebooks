# Os primeiros zeros são as coordenadas e os outros uns são a area da figura
ax = fig.add_axes([0, 0, 1, 1]) 

# Features
languages = ["English", "French", "Spanish", "Latin", "German"]
people = [100, 50, 1500, 400, 70]

plt.ylabel("NUMBER OF PEOPLE")
